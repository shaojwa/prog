tsc是intel从pentium开始引入的一个64位寄存器，tsc寄存器在gdb中是看不到的。用法如下：
```
#include <unistd.h>
#include <iostream>
using namespace std;

uint64_t tsc_now() {
  uint32_t low, high;
  asm volatile("rdtsc": "=a"(low), "=d"(high));
  return (uint64_t(low) | uint64_t(high) << 32);
}

int  main() {
  uint64_t start, end;
  start = tsc_now();
  sleep(10);
  end = tsc_now();
  cout << "duration "  << end - start << endl;
  return 0;
}
```
